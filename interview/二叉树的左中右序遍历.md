### 二叉树的左中右序遍历

```javascript
const binaryTree = {
  val: 1,
  left: {
    val: 2,
    left: {
      val: 4,
      left: null,
      right: null
    },
    right: {
      val: 5,
      left: null,
      right: null
    }
  },
  right: {
    val: 3,
    left: {
      val: 6,
      left: null,
      right: null
    },
    right: {
      val: 7,
      left: null,
      right: null
    }
  }
};

/**
 * 1.先序遍历：根左右
 */

const preOrder = root => {
  if (!root) return;
  console.log(root.val);
  preOrder(root.left);
  preOrder(root.right);
};

// preOrder(binaryTree);

/**
 * 2.中序遍历：左根右
 */

const inOrder = root => {
  if (!root) return;
  inOrder(root.left);
  console.log(root.val);
  inOrder(root.right);
};

// inOrder(binaryTree);

/**
 * 3.后序遍历：左右根
 */

const postOrder = root => {
  if (!root) return;
  postOrder(root.left);
  postOrder(root.right);
  console.log(root.val);
};

postOrder(binaryTree);
```
